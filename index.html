vectors.rf 
; Vector operations
(set prices [100.5 101.2 99.8 102.1])
(set volumes [1000 1500 800 2000])

; Element-wise multiplication
(* prices volumes)
; → [100500.0 151800.0 79840.0 204200.0]

; Aggregations
(avg prices)  ; → 100.9
(sum volumes) ; → 5300
query.rf 
; Create a table
(set trades
  (table
    (dict `sym`price`qty
      (list `AAPL`MSFT`NVDA
            [178.5 402.3 891.2]
            [100 200 150]))))

; VWAP calculation
(select
  (dict `sym`vwap
    (list sym (/ (sum (* price qty)) (sum qty))))
  trades (group sym))
timeseries.rf 
; Time-based aggregation
(set ticks
  (table
    (dict `time`price`vol
      (list (time [34200001 34200015])
            [100.10 100.15]
            [100 250]))))

; 1-second OHLCV bars
(select
  (dict `o`h`l`c`v
    (list (first price) (max price)
          (min price) (last price)
          (sum vol)))
  ticks (group (xbar 1000 time)))
