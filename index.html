; Generate sequences and ranges

Python 
import rayforce as rf
from rayforce import Vector, Table, Dict, Symbol

# Create typed vectors
prices = Vector.f64([178.5, 402.3, 891.2])
symbols = Vector.symbol(["AAPL", "MSFT", "NVDA"])

# Create a table from dict
trades = Table.from_dict({
    "sym": symbols,
    "price": prices,
    "qty": Vector.i64([100, 200, 150])
})

# Execute queries
result = trades.select(["sym", "price"]) \
               .where("qty > 100") \
               .group_by("sym")

# Convert to pandas DataFrame
df = result.to_pandas()
print(df)
#    sym   price
# 0  MSFT  402.3
# 1  NVDA  891.2
